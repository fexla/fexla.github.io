private Label CreateListLabel (){
   Label label = new Label ();
   label.AddToClassList ("ListLabel");
   label.RegisterCallback<ClickEvent> ((type) => {
      if (type.clickCount == 2){
         Debug.Log (label.text + "双击" + _dictionary.Types.Count);
      }
   });
   return label;
}
